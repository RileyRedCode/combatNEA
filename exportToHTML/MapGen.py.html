26       '''

28       Parameters: textMap:list, obstacles:list

30       Purpose: Constructor 
31       '''
32       def __init__(self, textMap=False, obstacles = False):
33           self.mapWidth = 20
34           self.halfWidth = self.mapWidth // 2
35           self.biomes = ["Plains", "Mountain", "HMountain", "Forest", "Water", "DWater"]#List of all biomes possible
36           #Probabilities that are tied to each terrain
37           self.probability = {False:{"Plains":10, "Mountain":6, "HMountain":2, "Forest":8, "Water":6, "DWater":4, False:0},
38                 "Plains":{"Plains":25, "Mountain":0, "HMountain":0, "Forest":15, "Water":5, "DWater":0, False:0},
39                 "Mountain":{"Plains":0, "Mountain":20, "HMountain":15, "Forest":15, "Water":0, "DWater":0, False:0},
40                 "HMountain":{"Plains":0, "Mountain":35, "HMountain":15, "Forest":0, "Water":0, "DWater":0, False:0},
41                 "Forest":{"Plains":15, "Mountain":18, "HMountain":0, "Forest":25, "Water":0, "DWater":0, False:0},
42                 "Water":{"Plains":15, "Mountain":0, "HMountain":0, "Forest":0, "Water":15, "DWater":10, False:0},
43                 "DWater":{"Plains":0, "Mountain":0, "HMountain":0, "Forest":0, "Water":30, "DWater":15, False:0}}
44           #Influences
45           self.conflictions = {False:{"Plains":0, "Mountain":0, "HMountain":0, "Forest":0, "Water":0, "DWater":0, False:0},
46                         "Plains":{"Plains":0, "Mountain":1, "HMountain":1, "Forest":0, "Water":0, "DWater":1, False:0},
47                         "Mountain":{"Plains":1, "Mountain":0, "HMountain":0, "Forest":0, "Water":1, "DWater":1, False:0},
48                         "HMountain":{"Plains":1, "Mountain":0, "HMountain":0, "Forest":1, "Water":1, "DWater":1, False:0},
49                         "Forest":{"Plains":0, "Mountain":0, "HMountain":1, "Forest":0, "Water":1, "DWater":1, False:0},
50                         "Water":{"Plains":0, "Mountain":1, "HMountain":1, "Forest":1, "Water":0, "DWater":0, False:0},
51                         "DWater":{"Plains":1, "Mountain":1, "HMountain":1, "Forest":1, "Water":0, "DWater":0, False:0}}
52           self.NPCs = pygame.sprite.Group()
53           if not textMap:
54               self.textMap = [[False for count in range(self.mapWidth)] for i in range(self.mapWidth)]#Establishes a plain map
55           else:
56               self.textMap = textMap
57           if not obstacles:
58               self.obstacles = []
59           else:

62           ''' 
63           Name: generate 
64           Parameters: None 
65           Returns: self.textMap:list, self.obstacles:list, self.NPCs:sprite group 
66           Purpose: This generates a map using soft constraint satisfaction and then generates obstacles. Constraint satisfaction  
67           involves finding the tile that creates the least conflicts in order to generate a somewhat natural map while still  
68           being random. This works by selecting random tiles in order to (prevent artifacting) and then choosing a random terrain  
69           which is influenced by the probabilities of surropunding tiles.   
70           '''
71       def generate(self):#Generates a map
72           success = False
73           while not success:
74               # This checks the cells of the map in a spiral fashion
75               rotations = 0
76               width = 2
77               length = 0
78               for count in range(self.mapWidth ** 2):  # Multiplies width by height to get count
79                   if length == width - 1:
80                       rotations += 1
81                       length = 0
82                   if rotations == 4:
83                       width += 2
84                       rotations = 0
85                   if rotations == 0:  # Top side
86                       cellX = (self.halfWidth) - (width // 2) + length
87                       cellY = (self.halfWidth) - (width // 2)
88                   elif rotations == 1:  # Right side
89                       cellX = (self.halfWidth) + (width // 2) - 1
90                       cellY = (self.halfWidth) - (width // 2) + length
91                   elif rotations == 2:  # Bottom side
92                       cellX = ((self.halfWidth) + (width // 2) - 1) - length
93                       cellY = (self.halfWidth) + (width // 2) - 1
94                   elif rotations == 3:  # Left side
95                       cellX = (self.halfWidth) - (width // 2)
96                       cellY = (((self.halfWidth) + (width // 2)) - 1) - length
97                   length += 1
98   
99                   probs = self.checkProbability(cellX, cellY)[0]
100                  if self.textMap[cellY][cellX] == False:  # If the cell is undefined
101                      bestType = False
102                      leastConflicts = 99
103  
104                      # This creates a list of ranges for each probability
105                      pool = 0
106                      probabilities = []
107                      count = 0
108                      stagger = 0
109                      for key in probs:
110                          if count != 0:
111                              pool += probs[key]
112                              probabilities.append([stagger, stagger + probs[key]])
113                          stagger += probs[key]
114                          count += 1
115  
116                      for count in range(2):
117                          num = random.randint(0, pool - 1)# gens a random number in the range of all probabilities combined
118                          # Selects the biome which the random number is in range of
119                          found = False
120                          count = 0
121                          while not found:
122                              if num in range(probabilities[count][0], probabilities[count][1]):
123                                  found = True
124                                  tTerrain = self.biomes[count]
125                              count += 1
126  
127                          self.textMap[cellY][cellX] = tTerrain
128                          conflicts = self.checkProbability(cellX, cellY)[1]
129                          if conflicts < leastConflicts:# if the terrain is more likely than the current most likely
130                              leastConflicts = conflicts
131                              bestType = tTerrain
132                      self.textMap[cellY][cellX] = bestType
133              success = self.biomeCheck()
134              if not success:#Clears map so that the algorithm can retry
135                  self.clearMap()
136          self.obstacleGen()
137          return self.textMap, self.obstacles, self.NPCs
138  
139      ''' 
140      Name: checkProbability 
141      Parameters: x:int, y:int 
142      Returns: probs:dictionary, conflicts:int 
143      Purpose: This function will calculate the probabilities using the surrounding tiles to the attached tile and calculate how many conflicts are generated. 
144      '''
145      def checkProbability(self, x, y):#Gets a pool of probabilities
146          area = 3
147          conflicts = 0
148          probs = {False:0, "Plains":0, "Mountain":0, "HMountain":0, "Forest":0, "Water":0, "DWater":0}#Dictionary of probability for each terrain based of surrounding terrain
149          for xC in range((x - area), (x + area)):#This will cycle the 3 cells to the left and the right of the cell aswell as the cell itself
150              for yC in range(y - area, y + area):
151                  tempX = xC%self.mapWidth
152                  tempY = yC%self.mapWidth
153                  conflicts += self.conflictions[self.textMap[y][x]][self.textMap[tempY][tempX]]
154                  for key in self.probability:
155                      probs[key] += self.probability[self.textMap[tempX][tempY]][key]#Adds the probabilities from this cell
156          return probs, conflicts
157  
158      ''' 
159      Name: biomeCheck 
160      Parameters: None 
161      Returns: safe:boolean 
162      Purpose: Ensures that there is at least a certain amount of each type of tile. 
163      '''
164      def biomeCheck(self):
165          biomeCount = {"Plains": 0, "Mountain": 0, "HMountain": 0,
166                        "Forest": 0, "Water": 0, "DWater": 0}
167          for y in self.textMap:
168              for x in y:
169                  for bi in biomeCount:
170                      if bi == x:
171                          biomeCount[bi] += 1
172          safe = True
173          for bi in biomeCount:
174              if biomeCount[bi] < 7:
175                  safe  = False
176          return safe

178      ''' 
179      Name: clearMap 
180      Parameters: None 
181      Returns: None 
182      Purpose: This clears the map and will only be called if a map is unsatisfactory.

184      def clearMap(self):
185          self.textMap = [[False for count in range(self.mapWidth)] for i in range(self.mapWidth)]

195              for y in range(self.mapWidth):

200      Parameters: None 
201      Returns: None

204      def makeimage(self):
205          self.imgMap = Image.new( 'RGB', ((self.mapWidth*TERRAIN_SIZE[0]),(self.mapWidth*TERRAIN_SIZE[0])), "red")
206          for x in range(self.mapWidth):
207              for y in range(self.mapWidth):
208                  self.imgMap.paste(self.textMap[y][x].img, (x*TERRAIN_SIZE[0], y*TERRAIN_SIZE[0]))
209          self.imgMap.save("map.png")

211      ''' 
212      Name: obstacleGen 
213      Parameters: None 
214      Returns: self.obstacles:list 
215      Purpose: This generates obstacles and also instantiates NPCs for any houses. 
216      '''
217      def obstacleGen(self):
218          used = []
219          c = 0
220          for count in range(3):
221              x = random.randint(0, self.mapWidth-1)
222              y = random.randint(0, self.mapWidth-1)
223              while self.textMap[y][x] == "DWater" or self.textMap[y][x] == "Water" or (x, y) in used:
224                  x = random.randint(0, self.mapWidth - 1)
225                  y = random.randint(0, self.mapWidth - 1)
226              self.NPCs.add(Monarch((x * TERRAIN_SIZE[0]) + (TILE_SIZE * 8),
227                                    (y * TERRAIN_SIZE[1]) + (TILE_SIZE * 8)))
228              self.obstacles.append((((x * TERRAIN_SIZE[0]), (y * TERRAIN_SIZE[1]), self.NPCs.sprites()[c].id), "House"))
229              c += 1
230              used.append((x, y))
231          for count in range(self.mapWidth//2):
232              ranX = random.randint(0, self.mapWidth-1)
233              ranY = random.randint(0, self.mapWidth-1)
234              while  self.textMap[ranY][ranX] != "Plains" or (ranX, ranY) in used:
235                  ranX = random.randint(0, self.mapWidth - 1)
236                  ranY = random.randint(0, self.mapWidth - 1)
237              used.append((ranX, ranY))
238              ranX = ranX*TERRAIN_SIZE[0]+TERRAIN_SIZE[0]//2
239              ranY = ranY*TERRAIN_SIZE[0]+TERRAIN_SIZE[0]//2
240              self.obstacles.append(((ranX, ranY), "Grave"))
241  
242          for count in range(self.mapWidth//2):
243              ranX = random.randint(0, self.mapWidth-1)
244              ranY = random.randint(0, self.mapWidth-1)
245              while  self.textMap[ranY][ranX] != "Forest" or (ranX, ranY) in used:
246                  ranX = random.randint(0, self.mapWidth - 1)
247                  ranY = random.randint(0, self.mapWidth - 1)
248              num = random.randint(1,2)
249              for count in range(0, num):
250                  ranXo = ranX*TERRAIN_SIZE[0]+((TERRAIN_SIZE[0]//3) * random.randint(1,2))
251                  ranYo = ranY*TERRAIN_SIZE[0]+((TERRAIN_SIZE[0]//3) * random.randint(1,2))
252                  self.obstacles.append(((ranXo, ranYo), "Bush"))
253          return self.obstacles

255      ''' 
256      Name: createObstacles 
257      Parameters: None 
258      Returns: None 
259      Purpose: Instantiates buildings for each obstacle in obstacles. 
260      '''
261      def createObstacles(self):
262          self.obstacleList = pygame.sprite.Group()
263          for obstacle in self.obstacles:
264              if obstacle[1] == "House":
265                  self.obstacleList.add(House(*obstacle[0]))
266              elif obstacle[1] == "Grave":
267                  self.obstacleList.add(Grave(*obstacle[0]))
268              elif obstacle[1] == "Bush":
269                  self.obstacleList.add(Bush(*obstacle[0]))
270  
271      ''' 
272      Name: finalise 
273      Parameters: None 
274      Returns: imgMap:image, obstacleList:sprite group 
275      Purpose: Carries out all the steps needed after the initial map creation to finish it. 
276      '''
277      def finalise(self):
278          self.text2Tile()
279          self.makeimage()
280          self.createObstacles()
281          return self.imgMap, self.obstacleList

300  ''' 
301  Name: Obstacle 
302  Purpose: This is a parent class used for obstacles which are objects the player and enemies cannot pass through. 
303  '''
304  class Obstacle(pygame.sprite.Sprite):
305  
306      ''' 
307      Name: __init__ 
308      Parameters: x:int, y:int, width:int, height:int, image:string, name:string 
309      Returns: None 
310      Purpose: Constructor 
311      '''
312      def __init__(self, x, y, width, height, image, name):
313          super().__init__()
314          self.type = name
315          self.mapX = x
316          self.mapY = y
317          self.width = width
318          self.height = height
319          self.image = pygame.image.load(image)
320          self.image = pygame.transform.scale(self.image, (self.width, self.height))
321          self.rect = self.image.get_rect()
322          self.rect.center = (x, y)
323  
324      ''' 
325      Name: __repr__ 
326      Parameters: None 
327      Returns: string 
328      Purpose: Used for debug to check obstacle positions are valid. 
329      '''
330      def __repr__(self):
331          return f"{self.mapX}, {self.mapY}, obstacle"

351  ''' 
352  Name: Grave 
353  Purpose: Special type of obstacle which spawns enemies if a player is too close 
354  '''
355  class Grave(Obstacle):
356  
357      ''' 
358      Name: __init__ 
359      Parameters: x:int, y:int 
360      Returns: None 
361      Purpose: Constructor 
362      '''
363      def __init__(self, x, y):
364          super().__init__(x, y, TILE_SIZE, TILE_SIZE, "Assets/skull.png", "Grave")
365          self.lastSpawn = pygame.time.get_ticks()
366  
367      def checkSpawn(self, playerPos):
368          close = False
369          for pos in playerPos:
370              hypotenuse = math.sqrt(((pos[0] - self.mapX)**2) + ((pos[1] - self.mapY)**2))
371              if hypotenuse < 600:
372                  close = True
373          if close:
374              if pygame.time.get_ticks() - self.lastSpawn > 2000:
375                  self.lastSpawn = pygame.time.get_ticks()
376                  return self.mapX + TILE_SIZE + (TILE_SIZE//2), self.mapY +TILE_SIZE+ (TILE_SIZE//2)
377          return False
