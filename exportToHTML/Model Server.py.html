14   ''' 
15   Name: recv_from_client 
16   Parameters: conn:connection, client_list:list 
17   Returns: None 
18   Purpose: Will send messages from one client to the other while also registering any important messages to the game loop. 
19   '''
20   def recv_from_client(conn,client_list):
21       while True:
22           data = conn.recv(1024)
23           if data:
24               instructions = data.decode().split("#")
25               packet = json.loads(instructions[0])
26               for client in client_list:
27                   if client != conn:
28                       client.send((json.dumps(packet)+"#").encode())
29                   else:
30                       if packet["command"] == "MOVE":
31                           players[conn].mapX, players[conn].mapY = packet["data"]["xPos"], packet["data"]["yPos"]
32   
33                       if packet["command"] == "ENEMYHIT":
34                           for enemy in serverEnemies:
35                               if enemy.id in packet["data"]["list"]:
36                                   enemy.takeDamage(packet["data"]["damage"])
37   
38                       if packet["command"] == "STARTCONFIRMATION":
39                           for c in confirmationList:
40                               if c[0] == client:
41                                   c[1] = True
42   
43                       if packet["command"] == "DEATHCONFIRMATION":
44                           players[client].confirm = True
45   
46                       if packet["command"] == "REVCONFIRMATION":
47                           players[client].confirm = False
48   
49                       if packet["command"] == "CONFIRMATION":
50                           for enemy in serverEnemies:
51                               if enemy.id == packet["data"]["id"]:
52                                   enemy.confirm = True
53   
54                       if packet["command"] == "TALKSTOP":
55                           players[client].endTalk(True)
56   
57                       if packet["command"] == "PAUSE":
58                           players[client].paused = True
59   
60                       if packet["command"] == "UNPAUSE":
61                           players[client].paused = False
62   
63                       if packet["command"] == "REVIVAL":
64                           for p in players:
65                               if players[p].id in packet["data"]["idList"]:
66                                   players[p].reviveSelf()
67   
68   ''' 
69   Name: send_to_client 
70   Parameters: client_list:list, packet:dictionary, identity:connection 
71   Returns: None 
72   Purpose: This sends messages to both or one client depending on if it has been specified with the identity parameter. 
73   '''
74   def send_to_client(client_list, packet, identity=False):
75       if identity:
76           for client in client_list:
77               if client == identity:
78                   client.send((json.dumps(packet) + "#").encode())
79       else:
80           for client in client_list:
81               client.send((json.dumps(packet) + "#").encode())

83   ''' 
84   Name: gameLoop 
85   Parameters: players dictionary, serverEnemies:list, client_list:list, confirmationList:list 
86   Returns: None 
87   Purpose: Will check if all players have entered the game loop and if so begin determining activities such as damage collision checks, npc movement and enemy activities. 
88   '''
89   def gameLoop(players, serverEnemies, client_list, confirmationList):
90       game = False
91       while not game:

103                  send_to_client(client_list, dpacket)
104

106              if result:
107                  for npc in serverNPCs:
108                      if npc == result:

110                  packet = {"command": "TALK", "data": {"id": result.id}}
111                  send_to_client(client_list, packet, player)
112  
113          packet = {"command": "EXPLOSIONDAMAGE", "data": []}
114          for explosion in serverExplosions:
115              data = explosion.update(players, True)
116              if data[1]:
117                  for i in data[1]:
118                      players[i["id"]].takeDamage(i["damage"], True)
119                      if players[i["id"]].health == 0 and not  players[i["id"]].dead:
120                          players[i["id"]].die(True)
121                          dpacket = {"command": "DIE", "data":players[i["id"]].id}
122                          send_to_client(client_list, dpacket)
123                      packet["data"] = i["damage"]
124                      send_to_client(client_list, packet, i["id"])
125  
126              if data[0] == "kill":
127                  serverExplosions.remove(explosion)
128                  explosion.kill()
129  
130  
131          # Determining the action for each NPC
132          npcActions = []

136                  npcActions.append(action)
137          packet = {"command": "NPCACTIONS", "data": []}

140          send_to_client(client_list, packet)

143          packet = {"command": "ENEMIES", "data": []}
144          playerPos = []

147                  playerPos.append([players[player].mapX, players[player].mapY])

164                  serverExplosions.add(Explosion(enemy.mapX, enemy.mapY, enemy.mapX, enemy.mapY))
165                  serverEnemies.remove(enemy)

168          for action in enemyActions:
169              if action != None:
170                  if action[1] == "ENEMYMOVE":
171                      packet["data"].append({"id":action[0],"action":"MOVE", "x":action[2], "y":action[3]})
172  
173                  elif action[1] == "ENEMYATTACK":
174                      packet["data"].append({"id":action[0], "action": "ATTACK"})
175  
176                  elif action[1] == "ENEMYDIE":
177                      packet["data"].append({"id":action[0], "action": "DIE"})
