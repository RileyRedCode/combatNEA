18   Purpose: Returns True or false depending on if the two rectangles given overlap. 
19   '''
20   def checkCollision(x1, x2, y1, y2, ox1, ox2, oy1, oy2):#Give the start and end points of each objects x and y axis
21   	if ((x1 <= ox1 < x2) or (x1 < ox2 <= x2) or (ox1 < x1 < ox2)) and ((y1 <= oy1 < y2) or (y1 < oy2 <= y2) or (oy1 < y1 < oy2)):

23   	else:
24   		return False

400      Name: menu 
401      Parameters: list:list, position:int, optionList:list 
402      Returns: list:list, position:int, boolean 
403      Purpose: This detects if the player has inputted to move or select an item in the list. If not then a list of images  
404      of each option is created with the one being hovered over being coloured in red. The boolean that is returned  
405      signifies if the player has or hasn't selected. 
406      '''
407  	def menu(self, list, position, optionList):
408  		keys = pygame.key.get_pressed()
409  		#confirm
410  		if keys[pygame.K_SPACE]:
411  			return list, position, True
412  		if self.cooldown:
413  			if pygame.time.get_ticks() - self.cooldown > 250:
414  				self.cooldown = False
415  		# Move up
416  		if keys[pygame.K_w] and position > 0 and not self.cooldown:
417  			position -= 1
418  			self.cooldown = pygame.time.get_ticks()
419  		# Move down
420  		if list:
421  			if keys[pygame.K_s] and position < len(list) - 1 and not self.cooldown:
422  				position += 1
423  				self.cooldown = pygame.time.get_ticks()
424  		# Creating options
425  		list = []
426  		for i in optionList:
427  			if len(list) == position:
428  				list.append(self.font.render(i, False, (255, 0, 0)))
429  			else:
430  				list.append(self.font.render(i, False, (255, 255, 255)))
431  		return list, position, False
