5    ''' 
6    Name: NPC 
7    Purpose: Can be talked to and will sell goods to the player.  
8    '''
9    class NPC(pygame.sprite.Sprite):
10   	#class variables
11   	idcount = 0
12   	directions = {1:"left", 2:"up", 3:"right", 4:"down"}
13   	''' 
14       Name: __init__ 
15       Parameters: x:int, y:int, width:int, height:int, image:string, spkImage:string, id:int 
16       Returns: None 
17       Purpose: Constructor 
18       '''
19   	def __init__(self, x, y, width, height, image, spkImage, id = False):
20   		super().__init__()
21   		#Variables
22   		self.width = width
23   		self.height = height
24   		self.mapX = x
25   		self.mapY = y
26   		self.activity = ["idol", pygame.time.get_ticks()]#[activity, time started]
27   		self.customers = []
28   		self.dialogue = {}
29   		self.firstChat = False
30   
31   		#id
32   		if id:
33   			self.id = id
34   		else:
35   			self.id = NPC.idcount
36   		NPC.idcount += 1
37   
38   		#Assets
39   		self.image = pygame.image.load(image)
40   		self.image = pygame.transform.scale(self.image,(self.width,self.height))
41   		self.rect = self.image.get_rect()
42   		self.rect.center = -400, -400
43   
44   		self.speakImage = pygame.image.load(spkImage)
45   		self.speakImage = pygame.transform.scale(self.speakImage, (800, 800))
46   
47   	''' 
48       Name: determineState 
49       Parameters: None 
50       Returns: self.id:int, self.mapX:int, self.mapY:int 
51       Purpose: Will swap states if enough time has passed and it is not talking then return it's position. 
52       '''
53   	def determineState(self):
54   		if self.activity[0] != "talking":
55   			# This swaps states
56   			if pygame.time.get_ticks() - self.activity[1] >= 2000 and self.activity[0] == "idol":
57   				direction = self.directions[random.randint(1, 4)]
58   				self.activity = ["moving", pygame.time.get_ticks(), direction]
59   			elif pygame.time.get_ticks() - self.activity[1] >= 1000 and self.activity[0] == "moving":
60   				self.activity = ["idol", pygame.time.get_ticks()]
61   
62   			# Action
63   			if self.activity[0] == "moving":
64   				if self.activity[2] == "left":
65   					self.mapX -= 2
66   				elif self.activity[2] == "up":
67   					self.mapY -= 2
68   				elif self.activity[2] == "right":
69   					self.mapX += 2
70   				elif self.activity[2] == "down":
71   					self.mapY += 2
72   				return self.id, self.mapX, self.mapY
73   
74   		return False
75   
76   	''' 
77       Name: addCustomer 
78       Parameters: customer:int 
79       Returns: None 
80       Purpose: Adds customer to the list of customers and updates the activity state. 
81       '''
82   	def addCustomer(self, customer):
83   		self.customers.append(customer)
84   		if self.activity[0] != "talking":
85   			self.activity[0] = "talking"
86   
87   	''' 
88       Name: removeCustomer 
89       Parameters: customer:int 
90       Returns: None 
91       Purpose: Removes the specfied customer to the list of customers and updates the activity state. 
92       '''
93   	def removeCustomer(self, customer):
94   		self.customers.remove(customer)
95   		if len(self.customers) == 0:
96   			self.activity[0] = "idol"
97   
98   	''' 
99       Name: getSpeaker 
100      Parameters: None 
101      Returns: self.speakImage:image, self.object, start:string 
102      Purpose: Returns the speakers image aswell as which dialogue node should be started from depending on wether the player has spoke to them before. 
103      '''
104  	def getSpeaker(self):
105  		if not self.firstChat:
106  			start = "firstMeet"
107  		else:
108  			start = "greetings"
109  		return self.speakImage, self.dialogue, start

136  Name: Vertex 
137  Purpose: Stores a singular piece of dialogue it's neighbours and any conditions. Designined to be a node in a dialogue Graph 
138  '''
139  class Vertex:
140  	''' 
141      Name: __init__ 
142      Parameters: name:string, dialogue:string, option:string 
143      Returns: None 
144      Purpose: Constructor 
145      '''
146  	def __init__(self, name, dialogue, option = False):
147  		self.name = name
148  		self.dialogue = dialogue
149  		self.neighbours = []
150  		self.option = option
151  		self.condition = False

154      Name: getValue 
155      Parameters: None 
156      Returns: self.dialogue:string

190  ''' 
191  Name: DialogueGraph 
192  Purpose: This is an important data structure used to store an NPC's dialogue. Each node stores its neighbours allowing for a clear traversal during a conversation. 
193  '''
194  class DialogueGraph:
195  
196  	''' 
197      Name: __init__ 
198      Parameters: None 
199      Returns: None 
200      Purpose: Constructor 
201      '''
202  	def __init__(self):
203  		self.nodes = []
204  
205  	''' 
206      Name: makeEdge 
207      Parameters: vertex1:string, vertex2:string 
208      Returns: None 
209      Purpose: Adds a neighbour to the relevant node. 
210      '''
211  	def makeEdge(self, vertex1, vertex2):
212  		vertex1, vertex2 = self.getNode(vertex1), self.getNode(vertex2)
213  		if vertex1 != "Node could not be found" and vertex2 != "Node could not be found":
214  
215  				vertex1.addNeighbour(vertex2)
216  
217  		else:
218  			print("Node could not be found. Operation failed.")
219  
220  	''' 
221      Name: deleteEdge 
222      Parameters: vertex1:string, vertex2:string 
223      Returns: None 
224      Purpose: Removes a neighbour from the relevant node. 
225      '''
226  	def deleteEdge(self, vertex1, vertex2):
227  		vertex1, vertex2 = self.getNode(vertex1), self.getNode(vertex2)
228  		if vertex1 != "Node could not be found" and vertex2 != "Node could not be found":
229  
230  			vertex1.removeNeighbour(vertex2)
231  			vertex2.removeNeighbour(vertex1)
232  
233  		else:
234  			print("Node could not be found. Operation failed.")
235  
236  	''' 
237      Name: deleteNode 
238      Parameters: vertex:string 
239      Returns: None 
240      Purpose: Removes all edges fron a Node before deleting it. 
241      '''
242  	def deleteNode(self, vertex):
243  		vertex = self.getNode(vertex)
244  		for value in self.nodes:
245  			if vertex in value.neighbours:
246  				value.removeNeighbour(vertex)
247  		self.nodes.remove(vertex)
248  
249  	''' 
250      Name: getNode 
251      Parameters: vertex:string 
252      Returns: None 
253      Purpose: Returns the relevant node if it could be found. 
254      '''
255  	def getNode(self, vertex):
256  		# Vertex will be the value of the node
257  		for v in self.nodes:
258  			if v.name == vertex:
259  				return v
260  		return "Node could not be found"
261  
262  	''' 
263      Name: makeNode 
264      Parameters: vertex:string, dialogue:string, option:string 
265      Returns: None 
266      Purpose: Instantiates a vertex class and appends it to the nodes list. 
267      '''
268  	def makeNode(self, vertex, dialogue, option = False):
269  		self.nodes.append(Vertex(vertex, dialogue, option))

272  ''' 
273  Name: constructDialogue 
274  Parameters: dic:dictionary 
275  Returns: d:object 
276  Purpose: Creates a dictionary then converts every single key in the dictionary into a vertex. It then establishes edges between any relevant nodes.  
277  '''
278  def constructDialogue(dic):
279  	d = DialogueGraph()
280  	#Creating nodes
281  	for key in dic:
282  		option = False
283  		if len(dic[key]) == 3:
284  			option = dic[key][2]
285  		d.makeNode(key, dic[key][0], option)
286  	#Creating Edges
287  	for key in dic:
288  		for e in dic[key][1]:
289  			d.makeEdge(key, e)
290  	return d
